<script>
  import fetch from '@/services/fetch'
  import { mapState, mapActions } from 'vuex'
  export default {
    name:'MediaList',
    data() {
      return {
        // 报告列表
        mediaPosList: [],
        isLoading: false,
        currentPage: 1,
        pageSize: 20,
        total: 0,
 
      }
    },
    created() {
      this.fetch_mediaPos_list();
    },
    methods: {
      ...mapActions([
      ]),
      fetch_mediaPos_list() {
        this.isLoading = true;
        return new Promise((resolve, reject) => {
          fetch.post('/mediaPos/getList', {
            page: this.currentPage,
            limit: this.pageSize
          }).then( res => {
            this.isLoading = false;
            if (res.ret === 0 && res.data) {
              this.mediaPosList = res.data;
              this.total = parseInt(res.total);
            }
            resolve(res);
          })
            .catch( err => {
              this.isLoading = false;
              reject(err);
            })
        })
 
      },
 
      // 格式化列表的状态(status)信息
      formatstatus(status){
        let result = {};
        switch( status) {
          case 0:
            result = {
              type: 'danger',
              text: '停用'
            };
            break;
 
          case 1:
            result = {
              type: 'success',
              text: '开启'
            };
            break;
 
 
        }
        return result;
      },
      OpeneditStatus(row){
        let _id=parseInt(row.id);
        let _status;
        let str;
 
        if(row.status === 1){
          _status = 0;
          str = "是否停用该点位?";
        }else{
          _status = 1;
          str = "是否启用该点位?";
        }
        this.$confirm(str, '修改状态', {
          confirmButtonText: '确定',
          cancelButtonText: '取消',
          type: 'warning'
        }).then(() => {
          this.editStatus(_id,_status);
        }).catch(() => {
 
        });
      },
 
      editStatus(id,status){
        return new Promise((resolve, reject) => {
          fetch.post('/mediaPos/editStatus', {
            id: id,
            status:status,
          }).then( res => {
            if (res.ret === 0) {
              this.fetch_mediaPos_list();
              this.$message({
                message: '点位状态修改成功',
                type: 'success'
              });
            }
            resolve(res);
          })
            .catch( err => {
              reject(err);
            })
        })
      },
      OpenDelete(row){
        let _id=parseInt(row.id);
 
        this.$confirm('是否删除此点位', '删除点位', {
          confirmButtonText: '确定',
          cancelButtonText: '取消',
          type: 'warning'
        }).then(() => {
          this.deletePos(_id);
        }).catch(() => {
 
        });
      },
      deletePos(id){
        return new Promise((resolve, reject) => {
          fetch.post('/mediaPos/del', {
            id: id,
          }).then( res => {
            if (res.ret === 0) {
              this.fetch_mediaPos_list();
              this.$message({
                message: '点位删除成功',
                type: 'success'
              });
            }
            resolve(res);
          })
            .catch( err => {
              reject(err);
            })
        })
      },
      // 处理分页操作
      handle_page_change(currentPage) {
        this.currentPage = currentPage;
        this.fetch_mediaPos_list();
      },
    },
    components:{}
  }
</script>
