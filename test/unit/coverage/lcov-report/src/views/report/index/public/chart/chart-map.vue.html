<template>
<div>
    <div>
        <div id="chart" ref="chart"></div>
    </div>
</div>
</template>
<script>
import echarts from "echarts";
import { pointData, geoCoordMap, convertData, chinaJson } from "./map";
 
export default {
  name: "chart-map",
  props: {
    data: {
      type: Array,
      required: false
    },
    mapMax: {
      type: Number,
      required: false
    },
 
  },
  data() {
    return {};
  },
  watch: {
    data: {
      handler: function(val, oldVal) {
        this.data = val;
        this.createChart();
      },
      deep: true
    }
  },
  methods: {
    randomValue() {
      return Math.round(Math.random() * 2000);
    },
 
    createChart() {
      echarts.registerMap("china", chinaJson);
 
      var options = {
        tooltip: {},
        visualMap: {
          min: 0,
          max: this.mapMax,
          left: "left",
          top: "bottom",
          text: ["高", "低"],
          seriesIndex: [1],
          inRange: {
            color: ["#e0ffff", "#006edd"]
          },
          calculable: true
        },
        geo: {
          map: "china",
          // roam: true,  禁用缩放
          label: {
            normal: {
              show: true,
              textStyle: {
                color: "#984c4c"
              }
            }
          },
          itemStyle: {
            normal: {
              borderColor: "rgba(0, 0, 0, 0.1)"
            },
            emphasis: {
              areaColor: null,
              shadowOffsetX: 0,
              shadowOffsetY: 0,
              shadowBlur: 20,
              borderWidth: 0,
              shadowColor: "rgba(0, 0, 0, 0.5)"
            }
          }
        },
        series: [
          {
            type: "scatter",
            coordinateSystem: "geo",
            data: convertData(pointData, geoCoordMap),
            symbolSize: 5,
            label: {
              normal: {
                formatter: "{b}",
                position: "right",
                show: false //是否显示点的位置名称
              },
              emphasis: {
                show: false //是否显示点的位置名称
              }
            },
            itemStyle: {
              normal: {
                color: "#F06C00"
              }
            }
          },
          {
            name: "地域",
            type: "map",
            geoIndex: 0,
            data: this.data
          }
        ]
      };
      this.$chart.setOption(options,true);
    }
  },
  mounted() {
    this.$chart = echarts.init(this.$refs.chart);
    this.createChart();
  },
  components: {}
};
</script>
