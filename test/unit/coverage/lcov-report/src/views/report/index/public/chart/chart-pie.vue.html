<template>
  <div>
    <div>
      <div id="chart" ref="chart"></div>
    </div>
  </div>
</template>
<script>
import echarts from "echarts";
 
export default {
  name: "chart-pie",
  props: {
    data: {
      type: Array,
      required: false
    }
  },
  data() {
    return {};
  },
  watch: {
    data: {
      handler: function(val, oldVal) {
        this.data = val;
        this.createChart();
      },
      deep: true
    }
  },
  methods: {
    createChart() {
      var option = {
        tooltip: {
          trigger: "item",
          formatter: "{b} : {c} ({d}%)"    //   {a} <br/>{b} : {c} ({d}%)
        },
        calculable : true,
        series: [
          {
            name: "访问来源",
            type: "pie",
            radius: ["55%","70%"],
            center: ["50%", "50%"],
            hoverAnimation:true,
            hoverOffset:3,
            label: {
                normal: {
                    formatter: '{b|{b}：}{per|{d}%}',
                    rich: {
                        b: {
                            fontSize: 16,
                            lineHeight: 18
                        },
                        per: {
                            // color: '#eee',
                            // backgroundColor: '#334455',
                            // padding: [1, 1],
                            height:18,
                        }
                    }
                }
            },
            labelLine:{
              normal:{
                length:10,
                lineStyle:{
                  type:'dotted',
                  width:2
                }
              }
              
            },
            data: this.data,
          }
        ]
      };
      this.$chart.setOption(option,true);
    }
  },
  mounted() {
    this.$chart = echarts.init(this.$refs.chart);
    this.createChart();
  },
};
</script>
 
<style scoped>
#chart {
  margin: 10px 0;
  height: 350px;
}
