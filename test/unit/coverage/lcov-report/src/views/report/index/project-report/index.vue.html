<script>
import { mapState, mapMutations, mapActions } from "vuex";
import _ from "lodash";
 
import projectReportContent from "./project-report-content";
 
import advertiseType from "@/views/report/index/public/sub-type/advertise-type";
import creativeData from "@/views/report/index/public/sub-type/creative-data";
import creativeGroup from "@/views/report/index/public/sub-type/creative-group";
import crowdData from "@/views/report/index/public/sub-type/crowd-data";
import dataTrend from "@/views/report/index/public/sub-type/data-trend";
import dataType from "@/views/report/index/public/sub-type/data-type";
import devicePercent from "@/views/report/index/public/sub-type/device-percent";
import mediaType from "@/views/report/index/public/sub-type/media-type";
import pointPosition from "@/views/report/index/public/sub-type/point-position";
import regionType from "@/views/report/index/public/sub-type/region-type";
 
export default {
  name: "project-report",
  data() {
    return {
      isloading: true,
      isloadingTrendData: true,
      isloadingDevicePercent: { pie: true, line: true },
      isloadingAdvertiserType: true,
      isloadingMediaType: true,
      isloadingRegionType: true,
      isloadingPos: true,
      isloadingCrowd: true,
      isloadingCreativeGroup: true,
      isloadingCreativeData: true,
 
      headerTitle: [
        "项目",
        "总预算",
        "消费",
        "展现量",
        "点击量",
        "CPM",
        "CTR",
        "CPC",
        "Leads",
        "CPL",
        "CR"
      ],
      projectData: {},
 
      //项目数据趋势
      trendTitle: [],
      projectDataTrend: {
        xAxisData: [],
        seriesData: [],
        multiaxis: false
      },
 
      //终端占比
      devicePercentData: {
        xAxisData: [],
        lineSeriesData: [],
        pieSeriesData: [],
        tabletData: []
      },
 
      //广告形式top5
      advertiseTypeData: {
        pieSeriesData: [],
        tabletData: []
      },
 
      //媒体top5
      mediaData: {
        pieSeriesData: [],
        tabletData: []
      },
 
      //地域top5
      regionData: {
        mapData: [],
        tableData: []
      },
 
      //点位top5
      pointPositionData: [],
 
      //人群top5
      crowdTitle: [],
      crowdGroupData: {
        xAxisData: [],
        lineSeriesData: [],
        tabletData: []
      },
 
      //创意组Top5
      creativeGroupTitle: [],
      creativeGroupData: {
        xAxisData: [],
        lineSeriesData: [],
        tabletData: []
      },
 
      //创意top10
      creativeTableData: []
    };
  },
  computed: {
    ...mapState({
      settings: state => state.report.settings.project, //每次获取settings都会执行此箭头函数
      project_report: state => state.report.project_report
    })
  },
 
  watch: {
    project_report: {
      handler: function(val) {
        this.fetch_get();
      },
      deep: true
    }
  },
 
  methods: {
    ...mapMutations("report", [
      "UPDATE_PROJECT_SETTING",
      "UPDATE_PROJECT_REPORT",
      "UPDATE_CURRENT_ROUTE"
    ]),
    ...mapActions("report", [
      "fetch_filter_data",
      "fetch_advertiser_data",
      "fetch_advertiser_trend_data",
      "fetch_device_percent_data",
      "fetch_advertiser_type_data",
      "fetch_media_type_data",
      "fetch_region_type_data",
      "fetch_point_position_data",
      "fetch_crowd_data",
      "fetch_creative_group_data",
      "fetch_creative_data"
    ]),
 
    getPosition(id) {
      return this.settings.findIndex(function(item) {
        return item.id == id;
      });
    },
 
    getChecked(id) {
      return this.settings.filter(item => item.id == id)[0].checked;
    },
 
    fetch_get() {
      this.updateProjectData({ currentPage: 1, limit: 5 });
      this.updateDataTrendType({ filterValue: "1" });
      this.updateDevicePercentType({ filterValue: "1", indexValue: "5" });
      this.updateDevicePercentIndex({
        filterValue: "1",
        indexValue: "5",
        name: "CPC"
      });
      this.updateAdvertiserType({ filterValue: "1" });
      this.updateMediaType({ filterValue: "1" });
      this.updateRegionType({ filterValue: "1" });
      this.updatePointPosition({ mediaValue: "", filterValue: "1" });
      this.updateCrowdData({ indexValue: "1", filterValue: "1" });
      this.updateCreativeGroup({ indexValue: "1", filterValue: "1" });
      this.updateCreativeData({ filterValue: "1" });
    },
 
    updateProjectData(value) {
      this.isloading = true;
      this.fetch_advertiser_data({
        page: value.currentPage,
        limit: value.limit
      }).then(res => {
        this.projectData = res;
        this.isloading = false;
      });
    },
 
    //更新state，并加载整个界面
    updateDateAndProject(value) {
      this.UPDATE_PROJECT_REPORT(value);
    },
 
    //更新数据趋势 类型
    updateDataTrendType(value) {
      //获取广告主数据趋势
      this.isloadingTrendData = true;
      this.fetch_advertiser_trend_data({ type: value.filterValue }).then(
        res => {
          if (res.data.length == 0) {
            this.$message.error("无法查询到广告主数据趋势");
            this.isloadingTrendData = false;
            return;
          }
 
          let project_list = res.data.project_list;
          let xAxisData = res.data.date_arrange;
          let seriesData = [];
 
 
          for (let item in project_list) {
            this.trendTitle.push(project_list[item].name); //得到 折线图标题
            seriesData.push({
              name: project_list[item].name,
              data: project_list[item].value
            });
          }
          this.projectDataTrend.xAxisData = xAxisData;
          this.projectDataTrend.seriesData = seriesData;
          this.isloadingTrendData = false;
 
          /**
           * 判断选择的广告主是几个，如果一个，则可以选多指标，并且折线图为双轴，
           * 如果多个，则只能选一个指标，折线图为单轴
           */
          if (this.project_report.filterValue.length == 1) {
            this.projectDataTrend.multiaxis = true;
          } else {
            this.projectDataTrend.multiaxis = false;
          }
        }
      );
    },
 
    //更新终端占比
    updateDevicePercentType(value) {
      this.isloadingDevicePercent.pie = true;
      this.fetch_device_percent_data({
        type: value.filterValue,
        index: value.indexValue
      }).then(res => {
        if (res.data.length == 0) {
          this.$message.error("无法查询到终端占比");
          this.isloadingDevicePercent.pie = false;
          return;
        }
 
        let pieSeriesData = [];
        let tabletData = [];
        for (let item in res.data) {
          pieSeriesData.push({
            name: res.data[item].name,
            value: res.data[item].value
          });
          tabletData.push({
            top: res.data[item].top,
            name: res.data[item].name,
            value: res.data[item].value,
            percent: res.data[item].percent
          });
        }
        this.devicePercentData.pieSeriesData = pieSeriesData;
        this.devicePercentData.tabletData = tabletData;
 
        this.isloadingDevicePercent.pie = false;
      });
    },
    updateDevicePercentIndex(value) {
      this.isloadingDevicePercent.line = true;
      this.fetch_device_percent_data({
        type: value.filterValue,
        index: value.indexValue
      }).then(res => {
        if (res.data.length == 0) {
          this.isloadingDevicePercent.line = false;
          return;
        }
 
        let xAxisData = [];
        let seriesData = [];
        for (let item in res.data) {
          xAxisData.push(res.data[item].name);
          seriesData.push(res.data[item].value);
        }
        this.devicePercentData.xAxisData = xAxisData;
        this.devicePercentData.lineSeriesData = [
          { name: value.name, data: seriesData }
        ];
 
        this.isloadingDevicePercent.line = false;
      });
    },
 
    //更新广告形式
    updateAdvertiserType(value) {
      this.isloadingAdvertiserType = true;
      this.fetch_advertiser_type_data({ type: value.filterValue }).then(res => {
        if (res.data.length == 0) {
          this.isloadingAdvertiserType = false;
          return;
        }
 
        let pieSeriesData = [];
        let tabletData = [];
        for (let item in res.data) {
          pieSeriesData.push({
            name: res.data[item].name,
            value: res.data[item].value
          });
          tabletData.push({
            top: res.data[item].top,
            name: res.data[item].name,
            cost: res.data[item].value,
            percent: res.data[item].percent
          });
        }
        this.advertiseTypeData.pieSeriesData = pieSeriesData;
        this.advertiseTypeData.tabletData = tabletData;
 
        this.isloadingAdvertiserType = false;
      });
    },
 
    //更新媒体列表
    updateMediaType(value) {
      this.isloadingMediaType = true;
      this.fetch_media_type_data({ type: value.filterValue }).then(res => {
        if (res.data.length == 0) {
          this.isloadingMediaType = false;
          return;
        }
 
        let pieSeriesData = [];
        let tabletData = [];
        for (let item in res.data) {
          if (value.filterValue == "1") {
            pieSeriesData.push({
              name: res.data[item].adform,
              value: res.data[item].cost
            });
          }
          if (value.filterValue == "2") {
            pieSeriesData.push({
              name: res.data[item].adform,
              value: res.data[item].pv
            });
          }
          if (value.filterValue == "3") {
            pieSeriesData.push({
              name: res.data[item].adform,
              value: res.data[item].click
            });
          }
          if (value.filterValue == "4") {
            pieSeriesData.push({
              name: res.data[item].adform,
              value: res.data[item].leads
            });
          }
 
          tabletData.push({
            top: res.data[item].top,
            adform: res.data[item].adform,
            cost: res.data[item].cost,
            pv: res.data[item].pv,
            click: res.data[item].click,
            ctr: res.data[item].ctr,
            cpc: res.data[item].cpc
          });
        }
        this.mediaData.pieSeriesData = pieSeriesData;
        this.mediaData.tabletData = tabletData;
 
        this.isloadingMediaType = false;
      });
    },
 
    //更新地域
    updateRegionType(value) {
      this.isloadingRegionType = true;
      this.fetch_region_type_data({ type: value.filterValue }).then(res => {
        if (res.data.length == 0) {
          this.$message.error("无法查询到地域数据");
          this.isloadingRegionType = false;
          return;
        }
 
        let mapData = [];
        let tableData = [];
 
        //将地域中的省名全部转换为两个或三个字，不必显示自治区等
        res.data = _.map(res.data, item => {
          let name = item.name;
          if (name.search(/黑龙江/i) != -1 || name.search(/内蒙古/i) != -1) {
            return {
              top: item.top,
              name: name.substr(0, 3),
              value: item.value,
              percent: item.percent
            };
          } else {
            return {
              top: item.top,
              name: name.substr(0, 2),
              value: item.value,
              percent: item.percent
            };
          }
        });
        for (let item in res.data) {
          mapData.push({
            name: res.data[item].name,
            value: res.data[item].value
          });
        }
        this.regionData.mapData = mapData;
        this.regionData.tableData = res.data.slice(0,5);
 
        let maxItem = _.max(res.data, function(item) {
          return  +item.value;
        });
 
        this.regionData.mapMax = +maxItem.value;
 
        this.isloadingRegionType = false;
      });
    },
 
    //更新点位
    updatePointPosition(value) {
      this.isloadingPos = true;
      this.fetch_point_position_data({
        type: value.filterValue,
        media: value.mediaValue
      }).then(res => {
        this.pointPositionData = res.data;
 
        this.isloadingPos = false;
      });
    },
 
    //更新人群列表
    updateCrowdData(value) {
      this.isloadingCrowd = true;
      this.fetch_crowd_data({
        type: value.filterValue,
        index: value.indexValue
      }).then(res => {
