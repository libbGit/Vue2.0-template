<template>
  <!-- 预警  -->
  <el-collapse class="el-collapse-warn" v-if="data.length">
    <el-collapse-item name="1" >
      <alert-item slot="title"
        :name='firstItem.name'
        :describe='firstItem.describe'
        @click-alert='onClick(firstItem)'
      />
      <div v-for="item in data">
        <alert-item
          :name='item.name'
          :describe='item.describe'
          @click-alert='onClick(item)'
        />
      </div>
    </el-collapse-item>
  </el-collapse>
</template>
<script>
  import fetch from '@/services/fetch'
  import alertItem from './alert-item'
  export default {
    name: 'alert-list',
    components:{
      alertItem
    },
    props: {
      param:{
        type:Object
      },
      /*id: {
        type: [Number,String],
      },*/
      url:{
        type: [String],
        required: true,
      }
    },
    data () {
      return {
        data:[]
      }
    },
    watch: {
      param: {
        handler: function (val, oldVal) {
          this.fetch_list();
        },
        deep: true
      }
    },
    computed:{
      firstItem(){
        if (this.data.length) {
          return this.data[0];
        }
      }
    },
    created () {
      this.fetch_list();
    },
    methods: {
      fetch_list () {
        return new Promise( (resolve, reject) => {
          return fetch.post(this.url, this.param).then( res => {
              if( res.ret == 0) {
                this.data = res.data || [];
                resolve();
              }
              else {
                reject( new Error(res.msg));
              }
            }).catch( err => {
              reject( err );
            })
        })
      },
      onClick (item) {
        this.$emit('onView',item)
      }
    }
  }
</script>
