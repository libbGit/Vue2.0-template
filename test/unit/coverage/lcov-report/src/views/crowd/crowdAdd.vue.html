<script>
import fetch from "@/services/fetch";
// import tree_group from '@/api/tree_group.json';
// import GroupTree from './group-create/group-tree'
import { mapState, mapActions } from "vuex";
import _ from "lodash";
import $ from "jQuery";
 
export default {
  name: "CrowdAdd",
  data() {
    return {
      dataAttr: "",
      crowdData: {
        name: "",
        validaty: "30",
        type: "1"
      },
      idOptions: [
        {
          id_type: 1,
          label: "Cookie"
        },
        {
          id_type: 2,
          label: "IDFA"
        },
        {
          id_type: 3,
          label: "IMEI"
        }
      ],
      tree_data: [],
      defaultProps: {
        children: "child",
        label: "label"
      },
 
      idType: 1,
      idFile: "",
      idMd5: false,
 
      redType: "clk",
      clkdate: "",
      sitedate: "",
      clkLevel: "adv",
      siteType: "url",
      site_keys: "",
      clkTree: [],
      make: 0,
 
      errMsg: "",
      errstatus: false
    };
  },
  created() {
    this.getLevelData();
    this.getTreeAttrData();
  },
  methods: {
    getTreeAttrData() {
      fetch.post("/group/tags").then(res => {
        this.tree_data = _.cloneDeep(res.data);
      });
    },
    getLevel() {
      this.clkTree = [];
      this.$refs.treeAttr.setCheckedKeys([]);
      this.$refs.treeClkIds.setCheckedKeys([]);
      this.getLevelData();
    },
    getLevelData() {
      fetch.post("/group/getAdNodes", { type: this.clkLevel }).then(res => {
        if (res.ret === 0 && res.data) {
          this.clkTree = res.data;
        }
      });
    },
    getIdFile(e) {
      this.idFile = e.target.files[0];
    },
    saveDir() {
      this.make = 0;
      this.save();
    },
    saveGen() {
      this.make = 1;
      this.save();
    },
    save() {
      var setData = _.cloneDeep(this.crowdData);
      if (setData.name.replace(/\s+/g, "") == "") {
        this.$message({
          message: "请填写人群包名称",
          type: "error"
        });
        return false;
      } else {
        this.errstatus = false;
      }
      setData.advertiser_id = this.ader_id;
      setData.make = this.make;
      if (setData.type == 1) {
        setData.tag_ids = this.$refs.treeAttr.getCheckedKeys().join(",");
        if (setData.tag_ids == "") {
          this.$message({
            message: "请选择属性定向标签",
            type: "error"
          });
          return false;
        } else {
          this.errstatus = false;
        }
      } else if (setData.type == 2) {
        setData.id_type = this.idType;
        setData.file = this.idFile;
        if (setData.file == "") {
          this.$message({
            message: "请选择上传文件",
            type: "error"
          });
          return false;
        } else {
          this.errstatus = false;
        }
        if (setData.id_type != 1) {
          if (this.idMd5) {
            setData.encryption = 1;
          } else {
            setData.encryption = 0;
          }
        }
      } else if (setData.type == 3) {
        setData.re_target_mode = this.redType;
        if (setData.re_target_mode == "clk") {
          if (this.clkdate == "" || this.clkdate == null) {
            this.$message({
              message: "请选择数据时间",
              type: "error"
            });
            return false;
          } else {
            this.errstatus = false;
          }
          setData.data_scope_start = this.clkdate[0];
          setData.data_scope_end = this.clkdate[1];
          setData.level = this.clkLevel;
          setData.ids = this.$refs.treeClkIds.getCheckedKeys().join(",");
          if (setData.ids == "") {
            this.$message({
              message: "请选择标签",
              type: "error"
            });
            return false;
          } else {
            this.errstatus = false;
          }
        } else if (setData.re_target_mode == "site") {
          if (this.sitedate == "" || this.sitedate == null) {
            this.$message({
              message: "请选择数据时间",
              type: "error"
            });
            return false;
          } else {
            this.errstatus = false;
          }
          setData.data_scope_start = this.sitedate[0];
          setData.data_scope_end = this.sitedate[1];
          setData.site_type = this.siteType;
          setData.site_keys = this.site_keys;
          if (setData.site_keys == "") {
            this.$message({
              message: "请填写内容",
              type: "error"
            });
            return false;
          } else {
            this.errstatus = false;
          }
        }
      }
      var formData = new FormData();
      $.each(setData, function(key, tar) {
        formData.append(key, tar);
      });
      //console.log("setData:", setData);
      var xhr = new XMLHttpRequest();
      xhr.open("POST", "/api/group/create");
      xhr.addEventListener("load", this.nfComplete, false);
      xhr.send(formData);
    },
    nfComplete(e) {
      var response = JSON.parse(e.target.responseText);
      if (response && response.ret == 0) {
        this.$router.push({ name: "CrowdList" });
      } else {
        this.$message({
          message: response.msg,
          type: "error",
          showClose: true
        });
      }
    },
    reset() {
      this.crowdData = {
        name: "",
        validaty: "30",
        type: "1"
      };
      this.$refs.treeAttr.setCheckedKeys([]);
      this.idType = 1;
      this.idFile = "";
      this.idMd5 = false;
      document.getElementById("file").value = "";
      this.redType = "clk";
      this.clkdate = "";
      this.clkLevel = "adv";
      this.$refs.treeClkIds.setCheckedKeys([]);
      this.sitedate = "";
      this.siteType = "url";
      this.site_keys = "";
    }
  },
  computed: {
    ...mapState("ad", ["ader_id"])
  },
  components: {
    // GroupTree
  }
};
</script>
