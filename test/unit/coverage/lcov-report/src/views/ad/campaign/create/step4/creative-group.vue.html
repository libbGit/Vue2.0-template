<template>
  <el-select v-model="value" placeholder="请选择创意组" size="mini" @change="handle_change" clearable :disabled='campaign_audit_status=="审核通过"' @clear='handle_clear'>
    <el-option
      v-for="item in creative_group_list"
      :key="item.id"
      :label="item.name"
      :value="item.id">
    </el-option>
  </el-select>
</template>
<script>
  import fetch from '@/services/fetch'
  import _ from "lodash";
  import { mapState, mapGetters, mapActions } from 'vuex'
  export default {
    name: 'creative_group',
    data () {
      return {
        creative_group_list: [],
        value: '',
      }
    },
    props: {
      posId : {
        type: String,
        required: true
      },
      posName : {
        type: String,
        required: true
      },
      projectId: {
        type: String,
        required: true
      },
      activityId: {
        type: [String,Number],
        required: true
      }
    },
    computed: {
      ...mapState('ad', [
        'brand_id',
        'product_id',
      ]),
      ...mapState('ad/campaign', [
        'ad_pos',
        'zone_info',
        'campaign_audit_status'
      ]),
      ...mapGetters('ad/campaign', [
        'ad_pos_selected',
      ]),
    },
    created() {
      this.fetch_creative_list();
      var zone = this.zone_info.find( item => {
        var result = item.pos_id == this.posId;
        return result;
      });
      if(zone) {
        this.value = zone.creative_group_id
      }
    },
    methods: {
      ...mapActions('ad/campaign', [
        'update_zone_info',
      ]),
      fetch_creative_list () {
        // /campaign/relation_creative
        fetch.post('/creative_group/creative_info', {
          pos_id: this.posId,
          activity_id:this.activityId,
          project_id: this.projectId,
          brand_id: this.brand_id,
          product_line_id: this.product_id == '-1' ? '' : this.product_id
        }).then( res => {
          if( res.ret == 0 ) {
            this.creative_group_list = res.data
          }
          else {
            console.error(res.msg);
          }
        }).catch( err => {
          console.log(err);
        })
      },
      handle_clear(){
        var zone_info = _.cloneDeep(this.zone_info);
        var index = zone_info.findIndex( item => item.pos_id == this.posId);
        if( index != -1  ) {
          zone_info.splice(index, 1)
          this.update_zone_info(zone_info);
        }
      },
      handle_change(value) {
        var zone_info = _.cloneDeep(this.zone_info);
        var index = zone_info.findIndex( item => item.pos_id == this.posId);
        if( index == -1  ) {
          zone_info.push( {
            creative_group_id: value,
            pos_id: this.posId
          })
        }
        else {
          zone_info.splice(index, 1, {
            creative_group_id: value,
            pos_id: this.posId
          })
        }
        zone_info = zone_info.filter( item => {
          return this.ad_pos_selected.indexOf(item.pos_id) > -1;
        })
        // console.log('change-zone_info:',zone_info)
        this.update_zone_info(zone_info);
      }
    }
  }
</script>
