<script>
  import { mapState, mapGetters, mapActions } from 'vuex'
  import SidebarTree from '@/views/ad/public/sidebar-tree'
  import ActivityNav from '@/views/ad/public/activity-nav'
 
  import CampaignSteps from '../public/campaign-steps'
  import CreativeGroup from './creative-group'
  import fetch from '@/services/fetch'
 
  import _ from 'lodash'
 
  export default {
    name: 'CampaignStep4',
    data() {
      return {
        rowdata: "",
        loading: false,
        form: {
          name: '',
          region: '',
          date1: '',
          date2: '',
          delivery: false,
          type: [],
          resource: '',
          desc: '',
        },
 
        popDiaCy: false,
        formLabelWidth: '90px',
        cyName:"",  //创意组名称
        cySeDataAry:[], //创意已选ID
        cytype: '1',
        setMedId: "",
        cyDataList:[], //穿梭框列表
        getCyDataList: [],//创意列表
        rankCyList: [], //穿梭选中排序列表
        cySeList:[],  //创意已选列表
        cyNumType: "1", //顺序轮播类型
        cyDz:"",      //定针
        cyNt3Num:"",  //自定义次数
        cyQzList: [], //权重列表
        errstatus: false,
        errMsg: ""
      }
    },
    computed: {
      ...mapState('ad',[
        'ader_id',
        'activity_id',
        'ader_id',
        'ader_name',
        'brand_id',
        'brand_name',
        'product_id',
        'product_name',
        'project_id',
        'project_name',
        'activity_id',
        'activity_name',
        'campaign_id',
        'campaign_name'
      ]),
      ...mapState('ad/campaign',[
        'ad_pos_list',
        'ad_pos',
        'zone_info',
        'campaign_audit_status'
      ]),
      ...mapGetters('ad/campaign', [
        'ad_pos_selected',
      ]),
      seleted_pos_list () {
        return this.ad_pos_list.filter( row => {
          return this.ad_pos_selected.indexOf(row.id) > -1
        })
      }
    },
    created () {
      this.update_step_index(3);
      // this.getCylist();
    },
    methods: {
      ...mapActions('ad/campaign', [
        'update_step_index',
        'create_campaign_step4',
        'push_campaign',
        'update_zone_info',
      ]),
      valid () {
        var _this = this;
        var zone_info = this.zone_info.filter( item => {
          return this.ad_pos_selected.indexOf(item.pos_id) > -1;
        })
        this.update_zone_info(zone_info);
        return new Promise( (resolve, reject) => {
          if( zone_info.length == 0 ) {
            reject( new Error('请至少为一个点位选择创意组'));
          }
          else {
            resolve();
          }
        })
      },
      handle_save (formName) {
        this.valid()
          .then( () => {
            this.create_campaign_step4()
              .then( res => {
                this.$message.success('计划保存成功');
                this.$router.push('/ad/campaign/manage');
              })
              .catch( err => {
                this.$message.error(err.message)
              })
          })
          .catch( err => {
            this.$message.error(err.message)
          })
      },
      handle_save_push (){
        this.valid()
          .then( () => {
            this.create_campaign_step4()
              .then( res => {
                this.push_campaign()
                  .then( () => {
                    this.$message.success('保存并推送PBS成功');
                    this.$router.push('/ad/campaign/manage');
                  })
                  .catch( err => {
                    this.$message.error(err.message)
                  })
              })
              .catch( err => {
                this.$message.error('保存计划失败', err.message);
              })
          })
          .catch( err => {
            this.$message.error(err.message)
          })
      },
      handle_cancel (formName) {
        this.$confirm('取消后你将会丢失创建的计划内容?', '提示', {
          type: 'warning'
        }).then(() => {
          this.$router.push('/ad/campaign/manage');
        })
 
      },
 
      //bug 9874
      handle_prev(){
        this.$router.push('step3');
      },
 
      open_PopCy(obj){
        var parms = {
          brand_id: this.brand_id,
          activity_id:this.activity_id,
          product_line_id: this.product_id || "",
          project_id: this.project_id,
          pos_id: obj.id
        };
        this.popDiaCy  = true;
        this.rowdata = obj;
        fetch.post('/creative_group/getCreative', parms).then( res => {
 
          if (res.ret === 0 && res.data) {
            this.cyName = "";
            this.getCyDataList = [];
            this.cyDataList = [];
            this.cySeDataAry = [];
            this.rankCyList = [];
            this.cySeList = [];
            this.cytype = "1";
            this.cyNumType = "1";
            this.cyDz = "";
            this.cyNt3Num = "";
            this.cyQzList = [];
            this.errstatus = false;
            this.errMsg = "";
            this.getCyDataList = _.cloneDeep(res.data);
            const data = [];
            const cylist = res.data;
            cylist.forEach((cy,index)=>{
              data.push({
                label: cy.name,
                key: cy.id,
                pinyin: cy.name
              })
            })
            this.cyDataList =  _.cloneDeep(data);
          }
        })
      },
 
      handleChange(value, direction, movedKeys) {
        const ary = [];
        const list = value;
        this.getCyDataList.forEach((obj)=>{
          list.find(function(value){
            if(value==obj.id){
              ary.push(obj);
            }
          })
        });
        this.cySeList = _.cloneDeep(ary);
        this.rankCyList = _.cloneDeep(ary);
      },
 
      touponeStep(i,tar){
        this.rankCyList.splice(i,1);
        this.rankCyList.splice((i-1),0,tar);
      },
 
      saveCy(){
        var setData = {};
        setData.brand_id = this.brand_id;
        setData.project_id = this.project_id;
        setData.activity_id = this.activity_id;
        setData.media_id = this.rowdata.media_id;
        setData.product_line_id = this.product_id;
        setData.pos_id = this.rowdata.id;
        setData.creative_id = this.cySeDataAry.join(",");
        setData.pattern = this.cytype;
        setData.name = this.cyName;
        if(setData.name.replace(/\s+/g, "")==""){
          this.errMsg = "请填写创意组名称！";
          this.errstatus = true;
          return false;
        }
        if(this.cySeList.length==0){
          this.errMsg = "请至少选择一项创意在保存！";
          this.errstatus = true;
          return false;
        }
        if(this.cytype==2){
          var type2Obj = {};
          if(this.cyNumType==1){
            type2Obj.type = 1;
            type2Obj.creative_id = [];
            this.rankCyList.forEach((obj)=>{
              type2Obj.creative_id.push(obj.id);
            });
            if(this.cyDz==""){
              this.errMsg = "请选择定针创意！";
              this.errstatus = true;
              return false;
            }
            type2Obj.frame = this.cyDz;
          }
          else if(this.cyNumType==2){
            type2Obj.type = 2;
            type2Obj.creative_id = [];
            this.rankCyList.forEach((obj)=>{
              type2Obj.creative_id.push(obj.id);
            });
          }
          else if(this.cyNumType==3){
            type2Obj.type = 3;
            type2Obj.creative_id = [];
            this.rankCyList.forEach((obj)=>{
              type2Obj.creative_id.push(obj.id);
            });
            var reg = new RegExp("^\\+?[1-9][0-9]*$");
            if(!reg.test(this.cyNt3Num)){
              this.errMsg = "请输入数字格式的自定义次数！";
              this.errstatus = true;
              return false;
            }
            if(this.cyDz==""){
              this.errMsg = "请选择定针创意！";
              this.errstatus = true;
              return false;
            }
            type2Obj.frame = this.cyDz;
            type2Obj.num = this.cyNt3Num;
          }
          setData.pattern_value = type2Obj;
        }
        else if(this.cytype==3){
          var type3Obj = {
            creative_id:[],
            priority:[]
          };
          var stat = false;
          var total = Number();
          this.cySeList.forEach((obj)=>{
            type3Obj.creative_id.push(obj.id);
            type3Obj.priority.push(obj.value);
            total+= Number(obj.value);
            if(!/^[0-9]*$/.test(obj.value)){
              stat = true;
            }
          });
          if(stat){
            this.errMsg = "请输入权重数值，不需要%！";
            this.errstatus = true;
            return false;
          }
          else if(total!=100){
            this.errMsg = "权重总和必须等于100！";
            this.errstatus = true;
            return false;
          }
          setData.pattern_value = type3Obj;
        }
        fetch.post("/creative_group/create",setData).then(res=>{
          if(res.ret === 0 && res.data){
            this.popDiaCy = false;
            var group = this.$refs['sGroup' + setData.pos_id]
            if (group && group.fetch_creative_list) {
              group.fetch_creative_list();
            }
          }
        })
      },
    },
    components: {
      SidebarTree,
      ActivityNav,
      CampaignSteps,
      CreativeGroup
    }
  }
</script>
